Template.filters.helpers({
  users: function() {
    var experiences = Experiences.find().fetch();
    var users = _.pluck(experiences, 'user');

    return _.uniq(users);
  }
});

Save the filter

Template.filters.events({
  'click li': function(e, tmpl) {
    var clickedUser = this;
    var selectedUser = Session.get('selectedUser');
    var newSelectedUser;

    if (newSelectedUser == clickedUser)
      newSelectedUser = ''; // Deselect
    else
      newSelectedUser = clickedUser; // Select new

    Session.set('selectedUser', newSelectedUser);
  }
});

Query should reflect filter

Template.listView.helpers({
  experiences: function() {
    var user = Session.get('selectedUser');
    var query = {};

    if (user)
      query.user = user;

    return Experiences.find(query);
  }
});

Figure out the query

Template.listView.helpers({
  experiences: function() {
    var query = {}, users = [], devices = [];

    Filters._collection.find({ selected: true, }).forEach(function(filter) {
      var type = filter.type, value = filter.value;

      if (type == 'user') {
          users.push(value);
      } else if (type == 'device') {
        devices.push(value);
      }

      if (users.length)
        query.user = {$in: users};
      if (devices.length)
        query.device = {$in: devices};
    });

    Session.set('query', query);
    return Experiences.find(query);
  }
});
